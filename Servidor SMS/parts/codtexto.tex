\chapter{Codificación utilizada en el texto}
\begin{description}
	\item[\reg{REG}] Se trata de un registro de 8 bits.
	\item[\reg{REG<n>}] Se trata del bit numero `n' del registro \reg{REG}.
	\item[\reg{REG<n1:n2>}] Se trata de un subconjunto de bits del registro \reg{REG}, siendo la concatenación (:) de los bits desde el bit numero `n1' al bit numero `n2' inclusive.
	\item[\bp{x:y}] Se trata de una concatenación desde un bit \bp{x} al bit \bp{y}.
	\item[b'xxxx1010'] Representa un valor binario en donde la `x' representa un valor indeterminado (entre `0' y `1') y los números los valores determinados.
	\item[0xHH] Representa un valor hexadecimal de 8\un{bits} en donde la `H' representa a un digito entre `0' a `F'.
	\item[HHh] Representa un valor hexadecimal de 8\un{bits} en donde la `H' representa a un digito entre `0' a `F'.
	\item[0xHHHH] Representa un valor hexadecimal de 16\un{bits} en donde la `H' representa a un digito entre `0' a `F'.
	\item[HHHHh] Representa un valor hexadecimal de 16\un{bits} en donde la `H' representa a un digito entre `0' a `F'.
\end{description}
